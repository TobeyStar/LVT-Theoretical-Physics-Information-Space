\section{Pre-Flight Checklist (Cold Boot)}
Executed before every mission is launched from dock.

\begin{enumerate}[label=\textbf{\arabic*.}]
    \item \textbf{Reactor Status:} Verify MSR temperature $> 650^\circ$C. Turbines at 100\% RPM.
    \item \textbf{Cryogenics:} Check SRC jacket temperature.
    \begin{itemize}
        \item \textit{Setpoint:} $< 4.2$ K (To minimize internal entropy).
        \item \textit{Action on deviation:} Abort launch. Risk of magnet quench and decoherence leakage.
    \end{itemize}
    \item \textbf{Informational Calibration ($C$):} Initiate sub-critical PFE-Ping.
    \begin{itemize}
        \item \textit{Purpose:} NavCom must measure the ship's exact informational complexity ($C$), including new cargo and crew, to calculate the charge requirement ($E_{iso} = \kappa_N \cdot C \cdot \ln(O^{-1})$).
    \end{itemize}
    \item \textbf{SMES Bank:} Verify superconducting status. (Charge = 0\% during pre-flight).
    \item \textbf{NavCom Diagnostics:} Run tests on IVM coils. Verify capacity for informational rotation.
    \item \textbf{Beacon Link:} Establish Handshake with the destination beacon.
    \begin{itemize}
        \item \textit{Signal Strength:} $> -80$ dBm.
        \item \textit{Encryption:} Target node address ($LV_{target}$) verified via QKD.
    \end{itemize}
\end{enumerate}

\section{Emergency Ops}

\subsection{RED ALERT: Topological Recoil (TRP Risk)}
\textit{Indicator:} Acoustic alarm. HUD flashes red. Active isolation ($E_{\mathrm{iso}}$) is over-excited, attempting to force decoherence ($O$) past the asymptotic safety margin. The matrix will resist and dump the excess energy back into the cavity.
\begin{enumerate}[label=\textbf{Step \arabic*.}]
    \item \textbf{SCRAM:} Strike the physical TRP emergency breaker (Hardware Override).
    \item \textbf{Dump:} SMES energy is dumped into external resistors (Expect an extremely white flash outside the hull when the charge is aborted).
    \item \textbf{Brace:} If the TRP fails to sever the connection, lethal ionizing radiation (Schwinger limit pair-production) will instantly flood the cabin.
\end{enumerate}



\subsection{YELLOW ALERT: QEC Failure (Quantization Noise)}
\textit{Indicator:} Warning: "Squeezed State Lost". Quantization noise from the matrix is penetrating the cavity, causing the rendering to drop data bits.
\begin{enumerate}[label=\textbf{Step \arabic*.}]
    \item \textbf{ABORT:} Abort the jump immediately if PFE has not fired and $(C \cdot O)$ is still $> T_{\mathrm{local}}$.
    \item \textbf{Shields:} If injection is underway (too late to abort), activate secondary magnetic radiation shielding.
    \item \textbf{Brace:} Prepare for a massive dose of informational radiation (Hawking-analogue) as the matrix drops atoms. Medical triage is required immediately upon arrival.
\end{enumerate}

\subsection{BLIND JUMP (The Last Resort)}
\textit{Scenario:} Hostile fire, collision course with unknown matter, or total system collapse. No Beacon available.
\begin{enumerate}[label=\textbf{Step \arabic*.}]
    \item \textbf{CMD \& PLT:} Turn both "Override Keys" simultaneously to lift the interlock against unknown nodes.
    \item \textbf{PLT:} Input manual pointer-guess.
    \item \textbf{CMD:} "Punch it."
    \item \textbf{Risk:} Extreme. Risk of rendering inside solid matter. Imminent risk of lethal shear stresses due to uncompensated $\Delta v$.
\end{enumerate}
